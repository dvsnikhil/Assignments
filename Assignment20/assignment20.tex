\item Show that points $A(a, b+c), B(b, c+a), C(c, a+b)$ are collinear.

\begin{enumerate}[label=(\roman*)]
\item $(k,0), (4,0), (0,2)$

\end{enumerate}
\item
\begin{enumerate}[label=(\roman*)]

\end{enumerate}
\item If area of triangle is 35 square units with vertices $(2,6), (5,-4)$ and $(k,4)$. Then $k$ is:
\begin{enumerate}
\item $12$
\item $-2$
\item $-12,-2$
\item $12, -2$
\end{enumerate}
\end{enumerate}
